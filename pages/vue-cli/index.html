# vue-cli

# vue文件编译过程

  • 首先webpack中给.vue格式的文件增加vue-loader处理模块；

  • 其次webpack增加vueLoaderPlugin插件；

  • vueLoaderPlugin的主要作用是将.vue文件中的各语言块应用在相应的rules上，并增加vueLoaderPitcher这个loader给.vue?vue格式的文件进行处理；

  • vueLoaderPitcher主要作用是拦截vue各模块，并给他们增加对应的loader处理模块；

// webpack.config.js
const path = require('path')
const VueLoaderPlugin = require('vue-loader/lib/plugin')

module.exports = {
    mode: 'development',
    module: {
        rules: [
            // ... 其它规则
            {
                test: /\.vue$/,
                loader: 'vue-loader'
            },
            // 它会应用到普通的 `.js` 文件，以及 `.vue` 文件中的 `<script>` 块
            {
                test: /\.js$/,
                loader: 'babel-loader'
            },
            // 它会应用到普通的 `.css` 文件，以及 `.vue` 文件中的 `<style>` 块
            {
                test: /\.css$/,
                use: [
                    'vue-style-loader',
                    'css-loader'
                ]
            },
            // 它会应用到普通的 `.sass` 文件，以及 `.vue` 文件中的 `<style lang="sass">` 块
            {
                test: /\.sass$/,
                use: [
                    'vue-style-loader',
                    'css-loader',
                    {
                        loader: 'sass-loader',
                        options: {
                            indentedSyntax: true  //sass-loader 默认解析 SCSS 语言
                        }
                    }
                ]
            }
        ]
    },
    plugins: [
        // 这个插件使 .vue 中的各类语言块匹配相应的规则
        new VueLoaderPlugin()
    ]
}

  1. App.vue原始文件

    }
